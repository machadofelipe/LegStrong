364 static inline void PI_run_parallel(PI_Handle handle,const float_t refValue,const float_t fbackValue,
365  const float_t ffwdValue,float_t *pOutValue)
366 {
367  float_t Error;
368  float_t Kp = PI_getKp(handle);
369  float_t Ki = PI_getKi(handle);
370  float_t Up;
371  float_t Ui = PI_getUi(handle);
372  float_t outMax = PI_getOutMax(handle);
373  float_t outMin = PI_getOutMin(handle);
374 
375  Error = refValue - fbackValue;
376 
377  Up = Kp * Error; // Compute the proportional output
378  Ui = MATH_sat(Ui + (Ki * Error),outMax,outMin); // Compute the integral output
379 
380  PI_setUi(handle,Ui);
381  PI_setRefValue(handle,refValue);
382  PI_setFbackValue(handle,fbackValue);
383  PI_setFfwdValue(handle,ffwdValue);
384 
385  *pOutValue = MATH_sat(Up + Ui + ffwdValue,outMax,outMin); // Saturate the output
386 
387  return;
388 } // end of PI_run_parallel() function

397 static inline void PI_run_series(PI_Handle handle,const float_t refValue,const float_t fbackValue,
398  const float_t ffwdValue,float_t *pOutValue)

400  float_t Error;

409  Error = refValue - fbackValue;
