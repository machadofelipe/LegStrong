218 static inline void TRAJ_run(TRAJ_Handle handle)
219 {
220  _iq targetValue = TRAJ_getTargetValue(handle);
221  _iq intValue = TRAJ_getIntValue(handle);
222  _iq error = targetValue - intValue;
223  _iq maxDelta = TRAJ_getMaxDelta(handle);
224  _iq minValue = TRAJ_getMinValue(handle);
225  _iq maxValue = TRAJ_getMaxValue(handle);
226 
227  // increment the value
228  intValue += _IQsat(error,maxDelta,-maxDelta);
229 
230  // bound the value
231  intValue = _IQsat(intValue,maxValue,minValue);
232 
233  // store the value
234  TRAJ_setIntValue(handle,intValue);
235 
236  return;
237 } // end of TRAJ_run() function
